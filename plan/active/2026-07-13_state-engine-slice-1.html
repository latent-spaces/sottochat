for agentic workers — REQUIRED SUB-SKILL: use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkboxes for tracking.

goal — Every visible session card and detail header shows an honest attention state (needs decision, plan review, waiting…) derived from a hybrid engine: deterministic transcript heuristics always, authoritative Claude Code hooks when the user installs a one-line snippet — with a receipt explaining every state.

architecture — A new pure state machine (src/attention.ts) consumes the existing MetaEvent stream plus new additive parser events (raw AskUserQuestion payloads, plan exits, queue ops) and optional hook signals POSTed to /hooks. The server resolves attention per session on each event and on a 30s tick, ships it on the session snapshot, and broadcasts changes. No model calls, no new dependencies, no persistence changes.

tech stack — Bun + TypeScript (server), plain JS no-build SPA (frontend), bun test with jsonl fixtures.

Global constraints (every task inherits these):
  • Never restart the owner's live instance on :3737. All manual verification runs a scratch instance: META_PORT=3947 META_OBSERVER_ENABLED=0 bun src/server.ts.

  • UI copy is lowercase; destructive actions name their object (clear chat, not reset).

  • Commits: one per task, message given in the task. Never add Co-Authored-By lines.
  • Fixtures record real transcript shapes; each fixture file header-comments the Claude Code version it was captured from (transcript format is officially unstable).

// untruncated head of a tool_result's text — signature detection needs more
// than summarizeToolResult's 100-char clip.
function toolResultText(content: unknown): string {
  if (typeof content === "string") return content.slice(0, 400);
  if (!Array.isArray(content)) return "";
  for (const b of content as Block[]) {
    if (b?.type === "text" && typeof b.text === "string") return b.text.slice(0, 400);
  }
  return "";
}

function parseAskQuestions(input: unknown): AskQuestion[] {
  if (!input || typeof input !== "object") return [];
  const qs = (input as { questions?: unknown }).questions;
  if (!Array.isArray(qs)) return [];
  const out: AskQuestion[] = [];
  for (const q of qs) {
    if (!q || typeof q !== "object") continue;
    const qq = q as Record<string, unknown>;
    if (typeof qq.question !== "string") continue;
    const optsRaw = Array.isArray(qq.options) ? qq.options : [];
    const options: AskQuestion["options"] = [];
    for (const o of optsRaw) {
      if (!o || typeof o !== "object") continue;
      const oo = o as Record<string, unknown>;
      if (typeof oo.label !== "string") continue;
      options.push({
        label: oo.label,
        ...(typeof oo.description === "string" ? { description: oo.description } : {}),
      });
    }
    out.push({
      question: qq.question,
      ...(typeof qq.header === "string" ? { header: qq.header } : {}),
      ...(typeof qq.multiSelect === "boolean" ? { multiSelect: qq.multiSelect } : {}),
      options,
    });
  }
  return out;
}

Step 3: implement src/attention.ts
// derives one attention state per session from the MetaEvent stream
// (+ optional hook signals, task 6). pure and deterministic: no model
// calls, no timers — resolveAttention takes `now` so tests own the clock.
// masterplan §4: `unknown` is first-class; there is deliberately no `done`.
import type { MetaEvent } from "./jsonl";

export type AttentionWord =
  | "running" | "needs decision" | "needs permission" | "plan review"
  | "waiting" | "idle" | "erred" | "unknown";

export type Attention = {
  state: AttentionWord;
  confidence: "hook" | "strong" | "weak";
  since: number;
  receipt?: { kind: string; toolUseId?: string; excerpt?: string; ts: number };
};

type Pending = { toolUseId: string; ts: number; excerpt: string };

export type AttentionTracker = {
  pendingAsks: Map<string, Pending>;
  pendingPlans: Map<string, Pending>;
  turnOpen: boolean;
  sawAnyEvent: boolean;
  queuedCount: number;
  lastActivityTs: number;
  lastStopTs: number;
};

export function createAttentionTracker(): AttentionTracker {
  return {
    pendingAsks: new Map(),
    pendingPlans: new Map(),
    turnOpen: false,
    sawAnyEvent: false,
    queuedCount: 0,
    lastActivityTs: 0,
    lastStopTs: 0,
  };
}

function clip(s: string, n: number): string {
  return s.length > n ? `${s.slice(0, n)}…` : s;
}

export function ingestAttentionEvent(t: AttentionTracker, ev: MetaEvent): void {
  t.sawAnyEvent = true;
  switch (ev.kind) {
    case "user_message":
      t.turnOpen = true;
      t.pendingAsks.clear();
      t.pendingPlans.clear();
      t.lastActivityTs = ev.ts;
      break;
    case "assistant_text":
    case "tool_use":
      t.turnOpen = true;
      t.lastActivityTs = ev.ts;
      break;
    case "ask_question": {
      const q = ev.questions[0];
      const excerpt = q ? clip(q.header ? `${q.header}: ${q.question}` : q.question, 80) : "question";
      t.pendingAsks.set(ev.toolUseId, { toolUseId: ev.toolUseId, ts: ev.ts, excerpt });
      t.lastActivityTs = ev.ts;
      break;
    }
    case "plan_exit":
      t.pendingPlans.set(ev.toolUseId, { toolUseId: ev.toolUseId, ts: ev.ts, excerpt: "plan submitted" });
      t.lastActivityTs = ev.ts;
      break;
    case "tool_result":
    case "question_answered":
    case "tool_rejected":
      t.pendingAsks.delete(ev.toolUseId);
      t.pendingPlans.delete(ev.toolUseId);
      t.lastActivityTs = ev.ts;
      break;
    case "stop":
      t.turnOpen = false;
      t.lastStopTs = ev.ts;
      break;
    case "queue_op":
      if (ev.op === "enqueue") t.queuedCount += 1;
      else t.queuedCount = Math.max(0, t.queuedCount - 1);
      break;
    case "mode_change":
      break; // corroboration only — unused in v1 resolution
  }
}

export type ResolveOpts = { now: number; alive: boolean; lastEventTs: number };

function oldest(m: Map<string, Pending>): Pending {
  let best: Pending | undefined;
  for (const p of m.values()) if (!best || p.ts < best.ts) best = p;
  return best!;
}

export function resolveAttention(t: AttentionTracker, opts: ResolveOpts): Attention {
  if (!t.sawAnyEvent) {
    return { state: "unknown", confidence: "weak", since: opts.now, receipt: { kind: "no_signals", ts: opts.now } };
  }
  if (t.queuedCount === 0 && t.pendingAsks.size > 0) {
    const p = oldest(t.pendingAsks);
    return {
      state: "needs decision", confidence: "strong", since: p.ts,
      receipt: { kind: "ask_question", toolUseId: p.toolUseId, excerpt: p.excerpt, ts: p.ts },
    };
  }
  if (t.queuedCount === 0 && t.pendingPlans.size > 0) {
    const p = oldest(t.pendingPlans);
    return {
      state: "plan review", confidence: "strong", since: p.ts,
      receipt: { kind: "plan_exit", toolUseId: p.toolUseId, excerpt: p.excerpt, ts: p.ts },
    };
  }
  // temporal states land in task 5; honest fallback until then.
  return { state: "unknown", confidence: "weak", since: opts.now, receipt: { kind: "unresolved", ts: opts.now } };
}
